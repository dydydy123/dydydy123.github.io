д»ЈзҗҶзұ»еңЁзЁӢеәҸиҝҗиЎҢж—¶жүҚиў«еҲӣе»әзҡ„д»ЈзҗҶж–№ејҸиў«жҲҗдёәеҠЁжҖҒд»ЈзҗҶгҖӮеңЁйқҷжҖҒд»ЈзҗҶдёӯпјҢд»ЈзҗҶзұ»жҳҜе·Із»Ҹе®ҡд№үеҘҪзҡ„пјҢеңЁзЁӢеәҸиҝҗиЎҢд№ӢеүҚе°ұе·Із»Ҹзј–иҜ‘е®ҢжҲҗгҖӮиҖҢеҠЁжҖҒд»ЈзҗҶжҳҜеңЁиҝҗиЎҢж—¶ж №жҚ®жҲ‘д»¬дј е…Ҙзҡ„ж–№жі•еҠЁжҖҒз”ҹжҲҗзҡ„гҖӮеҠЁжҖҒд»ЈзҗҶзӣёиҫғдәҺйқҷжҖҒд»ЈзҗҶзҡ„дјҳеҠҝеңЁдәҺеҸҜд»ҘеҫҲж–№дҫҝзҡ„еҜ№д»ЈзҗҶзұ»зҡ„жүҖжңүеҮҪж•°иҝӣиЎҢз»ҹдёҖз®ЎзҗҶпјҢеҒҮеҰӮжҲ‘д»¬жғіеңЁжҜҸдёӘд»ЈзҗҶж–№жі•дёӯйғҪеҠ дёҖдәӣзӣёеҗҢд»Јз ҒпјҢеҰӮжһңд»ЈзҗҶж–№жі•еҫҲеӨҡпјҢйӮЈд№ҲжҲ‘д»¬йңҖиҰҒеңЁжҜҸдёӘд»ЈзҗҶж–№жі•дёӯеҶҷйҮҚеӨҚд»Јз ҒгҖӮ

еңЁjavaзҡ„java.lang.reflectеҢ…дёӢжҸҗдҫӣдәҶдёҖдёӘProxyзұ»е’ҢдёҖдёӘInvocationHandlerжҺҘеҸЈпјҢйҖҡиҝҮиҝҷдёӘзұ»е’ҢиҝҷдёӘжҺҘеҸЈеҸҜд»Ҙз”ҹжҲҗJDKеҠЁжҖҒд»ЈзҗҶзұ»е’ҢеҠЁжҖҒд»ЈзҗҶеҜ№иұЎгҖӮ

RentalProxyзұ»(д»ЈзҗҶзұ»)пјҡ

package Proxy2;

import java.lang.reflect.InvocationHandler;
import java.lang.reflect.Method;
import java.lang.reflect.Proxy;

public class RentalProxy {
    public static Rental getProxy(Rental obj){
 /**
 * public static Object newProxyInstance(ClassLoader loader,Class<?>[]  interfaces,InvocationHandler h)
 * еҸӮж•°дёҖ: зұ»еҠ иҪҪеҷЁпјҢиҙҹиҙЈеҠ иҪҪд»ЈзҗҶ
 * еҸӮж•°дәҢ: иҺ·еҸ–иў«д»ЈзҗҶеҜ№иұЎе®һзҺ°зҡ„е…ЁйғЁжҺҘеҸЈгҖӮд»ЈзҗҶиҰҒдёәе…ЁйғЁжҺҘеҸЈзҡ„е…ЁйғЁж–№жі•иҝӣиЎҢзҡ„д»ЈзҗҶ
 * еҸӮж•°дёү: д»ЈзҗҶзҡ„ж ёеҝғеӨ„зҗҶйҖ»иҫ‘
 */
        return (Rental) Proxy.newProxyInstance(obj.getClass().getClassLoader(), obj.getClass().getInterfaces(),
                new InvocationHandler() {
                    @Override
                    //еҸӮж•°дёҖ: д»ЈзҗҶеҜ№иұЎжң¬иә«
                    //еҸӮж•°дәҢ: жӯЈеңЁиў«д»ЈзҗҶзҡ„ж–№жі•
                    //еҸӮж•°дёү: иў«д»ЈзҗҶж–№жі•пјҢеә”иҜҘдј е…Ҙзҡ„еҸӮж•°
                    public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
                        System.out.println("дёӯд»Ӣд»Ӣе…Ҙ");
                        method.invoke(obj,args);
                        return null;
                    }
                });
    }
}
